-- Ghost Logic
local function manageGhost(state)
    if state then
        local char = LocalPlayer.Character
        if char then
            char.Archivable = true
            ghostModel = char:Clone()
            char.Archivable = false
            for _, v in pairs(ghostModel:GetDescendants()) do
                if v:IsA("BasePart") then
                    v.Anchored = true
                    v.CanCollide = false
                    v.Transparency = 0.5
                elseif v:IsA("LocalScript") or v:IsA("Script") then
                    v:Destroy()
                end
            end
            local head = ghostModel:FindFirstChild("Head")
            if head then
                local bb = Instance.new("BillboardGui", head)
                bb.Size = UDim2.new(0, 120, 0, 40)
                bb.AlwaysOnTop = true
                bb.ExtentsOffset = Vector3.new(0, 3, 0)
                local tl = Instance.new("TextLabel", bb)
                tl.Size = UDim2.new(1, 0, 1, 0)
                tl.Text = "SERVER POS"
                tl.TextColor3 = Color3.new(1, 0, 0)
                tl.BackgroundTransparency = 1
                tl.Font = Enum.Font.GothamBold
                tl.TextSize = 14
            end
            ghostModel.Parent = workspace
        end
    else
        if ghostModel then ghostModel:Destroy() ghostModel = nil end
    end
end

-- 4. BUTTONS & EVENT-BASED EXECUTION
local DesyncBtn = Instance.new("TextButton", Main)
DesyncBtn.Size = UDim2.new(0, 170, 0, 35)
DesyncBtn.Position = UDim2.new(0.5, -85, 0.45, 0)
DesyncBtn.Text = "DESYNC: OFF"
DesyncBtn.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
DesyncBtn.TextColor3 = Color3.new(1, 1, 1)
DesyncBtn.Font = Enum.Font.GothamBold
Instance.new("UICorner", DesyncBtn)

DesyncBtn.MouseButton1Click:Connect(function()
    getgenv().DesyncToggle = not getgenv().DesyncToggle
    
    -- Single-shot execution
    if getgenv().DesyncToggle then
        pcall(function() raknet.desync(true) end)
        DesyncBtn.Text = "DESYNC: ON"
        DesyncBtn.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
        StatusLabel.Text = "Status: DESYNCED"
        StatusLabel.TextColor3 = Color3.new(1, 0, 0)
    else
        pcall(function() raknet.desync(false) end)
        DesyncBtn.Text = "DESYNC: OFF"
        DesyncBtn.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
        StatusLabel.Text = "Status: Normal"
        StatusLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
    end
    
    manageGhost(getgenv().DesyncToggle)
end)

-- 5. UNIVERSAL DRAGGABLE LOGIC
local dragToggle, dragStart, startPos

Main.InputBegan:Connect(function(input)
    if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then
        dragToggle = true
        dragStart = input.Position
        startPos = Main.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then dragToggle = false end
        end)
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) and dragToggle then
        local delta = input.Position - dragStart
        local position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        TweenService:Create(Main, TweenInfo.new(0.12, Enum.EasingStyle.Sine), {Position = position}):Play()
    end
end)
